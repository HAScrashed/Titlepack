#Include "Libs/miltant/ObstacleV2/Titlepack/Menus/Windows.Script.txt"			as Windows
#Include "Libs/miltant/ObstacleV2/ManialinkLib.Script.txt"						as Styles
#Include "Libs/miltant/ObstacleV2/Titlepack/Menus/SubMenu/Index.Script.txt"		as SubMenu
#Include "Libs/miltant/ObstacleV2/Titlepack/Dialogs/Explorer/Index.Script.txt"	as ExplorerDialog


Text GetName() {
	return "editors";
}

Void Load() {
	ExplorerDialog::Load();

	declare SubMenu::K_SubMenu[] SubMenus;

	declare SubMenu::K_SubMenu Maps;
	Maps.Name = "Maps";
	Maps.PreventAutoNavigation = True;
	Maps.Image = "file://Media/Images/Obstacle/Moods/none.png";
	SubMenus.add(Maps);

	declare SubMenu::K_SubMenu Replay;
	Replay.Name = "Replay";
	Replay.PreventAutoNavigation = True;
	Replay.Image = "file://Media/Images/Obstacle/Moods/day.png";
	SubMenus.add(Replay);

	declare SubMenu::K_SubMenu Item;
	Item.Name = "Item";
	Item.PreventAutoNavigation = True;
	Item.Image = "file://Media/Images/Obstacle/Moods/night.png";
	SubMenus.add(Item);

	declare SubMenu::K_SubMenu Action;
	Action.Name = "Action";
	Action.PreventAutoNavigation = True;
	Action.Image = "file://Media/Images/Obstacle/Moods/none.png";
	SubMenus.add(Action);

	declare SubMenu::K_SubMenu Interface;
	Interface.Name = "Interface";
	Interface.PreventAutoNavigation = True;
	Interface.Image = "file://Media/Images/Obstacle/Moods/sunset.png";
	SubMenus.add(Interface);

	declare SubMenu::K_SubMenu Ghost;
	Ghost.Name = "Ghost";
	Ghost.PreventAutoNavigation = True;
	Ghost.Image = "file://Media/Images/Obstacle/Moods/none.png";
	SubMenus.add(Ghost);


	declare Windows::K_Window Window;
	Window.Name = GetName();
	Window.Manialink = SubMenu::Build(SubMenus);
	Window.Animation = CUILayer::EUILayerAnimation::LeftSlow;

	Windows::CreateWindow(Window);
}

Void Loop() {
	foreach(Event in PendingEvents) {
		if ( Event.Type == CManiaAppEvent::EType::LayerCustomEvent &&
			 Event.CustomEventData.count > 0 &&
			 Event.CustomEventType == "window_load") {
			switch (Event.CustomEventData[0]) {
				case "Maps": {
					declare Map = ExplorerDialog::Pick("Pick a map", "Map");
					if (Map != "") {
						TitleControl.EditMap(
							Map, 
							"MapTypes/ShootMania/ObstacleTitleArena.Script.txt",
							"""<mode_script_settings>
								<setting name='S_Mode'					value=''		type='text'/>
								<setting name='S_IsOfficialMap'			value='False'	type='boolean'/>
								<setting name='S_Gameplay_Acceleration'	value='1.'		type='real'/>
								<setting name='S_Gameplay_Control'		value='1.'		type='real'/>
								<setting name='S_Gameplay_Gravity'		value='1.'		type='real'/>
							</mode_script_settings>"""
						);
					}
				}
				case "Replay": {
					declare Replays = ExplorerDialog::MultiPick("Pick one or several replays", "Replay");
					if (Replays.count > 0)
						TitleControl.EditReplay(Replays);
				}
				case "Item": {
					TitleControl.OpenEditor(CTitleControl::EEditorType::ItemEditor);
				}
				case "Action": {
					TitleControl.OpenEditor(CTitleControl::EEditorType::ActionMaker);
				}
				case "Interface": {
					TitleControl.OpenEditor(CTitleControl::EEditorType::InterfaceDesigner);
				}
				case "Ghost": {
					TitleControl.OpenEditor(CTitleControl::EEditorType::InterfaceDesigner);
				}
			}

			yield;
		}
	}
}
